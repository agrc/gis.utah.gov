export const dateFormatter = Intl.DateTimeFormat('en-US', { dateStyle: 'full' });
export const res = await fetch('https://api.github.com/repos/agrc/api-client/releases?per_page=1');
export const data = await res.json();

export const publishedDate = new Date(data[0].published_at);
export const windowsLink = data[0].assets.find((asset) => asset.name.includes('win32-setup.exe')).browser_download_url;
export const macOsLink = data[0].assets.find((asset) => asset.name.includes('x64.dmg')).browser_download_url;

export const links = [
  {
    title: 'UGRC API Client for Windows',
    url: windowsLink,
  },
  {
    title: 'UGRC API Client for MacOS',
    url: macOsLink,
  },
];

<Layout title={frontmatter.title} subTitle={dateFormatter.format(publishedDate)} category={frontmatter.category} downloadSectionTitle="Get the app">
  <Metadata slot="metadata" {...frontmatter} />

  <ApplicationLink slot="application-link" href={frontmatter.application} />

  <DirectDownloads title="Install links" slot="downloads" links={links} />

  <Fragment slot="summary">
      The UGRC API Client is a cross-platform, stand-alone desktop geocoding tool designed to carefully guide you step-by-step through your geocoding tasks. This app is our most user friendly geocoding client for geocoding `*.csv` files. It does not require a license or subscription to use and automatically downloads updates as soon as they are available to stay current and add new features.
  </Fragment>

  <Fragment slot="description">
    The API Client is a great tool for users who are not familiar with the programming languages to interact with the API directly and want a user friendly interface to geocode their files.

    ## Getting started

    <div class="mx-auto w-full max-w-4xl">
      <YouTube
        class="aspect-video max-w-full rounded-lg border-2 border-black shadow-xl"
        id="BSmQ_9E0cVE"
        playlabel="Listen to Jake Adams explain how to use the UGRC API Client."
      />
    </div>

    To get started, download the API Client for the operating system that you are running below. MacOS and windows are the only supported operating systems at this time. Once you have downloaded the application, you will be instructed to create an account on the [API self service website](https://developer.mapserv.utah.gov). Be sure to verify the email address you sign up with! Once you have created an account, you will be able to generate an API key that you will use to authenticate with the API Client. You are now ready to geocode your `*.csv` files. The API Client will guide you through the process of geocoding your file and will creating a report of the results that you can join back to your data.
  </Fragment>
